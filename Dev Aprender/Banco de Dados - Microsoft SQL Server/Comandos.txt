Script01.sql
SELECT

SERVE PARA SELECIONAR UMA COLUNA OU MAIS DE UMA TABELA

ASSIM ESTÁ SELECIONANDO APENAS UMA COLUNA DE UMA TABELA
SELECT <nome da coluna> FROM <nome da tabela>

ASSIM ESTÁ SELECIONANDO TODAS AS COLUNA DE UMA TABELA
SELECT * FROM <nome da tabela>



Script02.sql
DISTINCT

SERVE PARA QUANDO VOCÊ QUER OMITIR OS DADOS DUPLICADOS DE UMA TABELA, PARA QUE VOCÊ RETORNE APENAS OS DADOS ÚNICOS

SELECT DISTINCT <nome da coluna> FROM <nome da tabela>



Script03.sql
WHERE

SERVE PARA QUANDO É NECESSÁRIO PEGAR APENAS ALGUNS DADOS DA TABELA

SELECT <coluna1, coluna2, coluna3> FROM <nome da tabela> WHERE <condição>



Script04.sql
COUNT

SERVE PARA RETORNAR O NÚMERO DE LINHAS QUE BATE COM A CONDIÇÃO QUE FOI DEFINIDA

SELECT COUNT(<nome da coluna> ou *) FROM <nome da tabela>
